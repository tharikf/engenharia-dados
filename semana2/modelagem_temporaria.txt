Table FatoVenda {
  ID_order serial [PK]
  ID_data date
  ID_ship_date date
  ID_ship_mode varchar
  Customer_ID int
  Product_ID varchar
  Localizacao_ID int
  Sales numeric

  Note: "Fato contendo a receita por pedido"
}

Table DimData {
  ID_data date [PK]
  Dia int
  Mes int
  Ano int
  Dia_Semana varchar
}

Table DimShipMode {
  ID_ship_mode varchar [PK]
  Nome_ShipMode varchar
}

Table DimCliente {
  Customer_ID int [PK]
  Customer_Name varchar
  Segment varchar
}

Table DimProduto {
  Product_ID varchar [PK]
  Product_Name varchar
  Sub_Category varchar
  Category varchar
}

Table DimLocalizacao {
  Localizacao_ID int [PK]
  Country varchar
  City varchar
  State varchar
  Postal_Code int
  Region varchar
}

Ref: FatoVenda.ID_data > DimData.ID_data
Ref: FatoVenda.ID_ship_date > DimData.ID_data
Ref: FatoVenda.ID_ship_mode > DimShipMode.ID_ship_mode
Ref: FatoVenda.Customer_ID > DimCliente.Customer_ID
Ref: FatoVenda.Product_ID > DimProduto.Product_ID
Ref: FatoVenda.Localizacao_ID > DimLocalizacao.Localizacao_ID
